Akcja 1 Dodanie nowych stron do nawigacji - sposób 1

Akcja tworzy kilka obiektów klasy NavigationPage a następnie dodaje je do nawigacji. 
    // Tworzymy obiekt NavigationPage...
    $newPage = new NavigationPage(
        array(
                'id' => 'firma',
                'parentId' => 'root',
                'label' => 'O firmie',
                'url' => '#'
        ));
    
    // ..i dodajemy go do nawigacji używając metody w klasie Navigation
    $navigation->addNavigationPage($newPage);

    
    // Powtarzamy kilka razy...
    $newPage2 = new NavigationPage(
        array(
                'id' => 'firma_historia',
                'parentId' => 'firma',
                'label' => 'Historia firmy',
                'url' => '#'
        ));
    $navigation->addNavigationPage($newPage2);

    $newPage3 = new NavigationPage(
        array(
                'id' => 'firma_zarzad',

                'label' => 'Zarząd firmy',

    $navigation->addNavigationPage($newPage3);

    $newPage4 = new NavigationPage(
